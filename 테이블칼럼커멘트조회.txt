SELECT 
	TABLE_NAME
	,TAB_COMMENTS
	--,COLUMN_ID
	,COLUMN_NAME
	,DATA_TYPE
	,NULLABLE
	,COMMENTS
FROM(
SELECT
	A.TABLE_NAME
	,C.COMMENTS AS TAB_COMMENTS
	,A.COLUMN_ID
	,A.COLUMN_NAME
	,A.DATA_TYPE
	,A.NULLABLE
	,B.COMMNETS
FROM	ALL_TAB_COMMENTS A
	,ALL_COL_COMMENTS B
	,ALL_TAB_COMMENTS C
WHERE	A.OWNER = B.OWNER
   AND	A.OWNER = C.OWNER
   AND	A.TABLE_NAME = B.TABLE_NAME (+)
   AND	A.TABLE_NAME = C.TABLE_NAME (+)
   AND	A.COLUMN_NAME = B.COLUMN_NAME (+)
   AND	A.OWNER = 'USER_AM'
   AND	A.TABLE_NAME NOT LIKE 'TAB' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'PLAN_TABLE' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'MLOG$' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'QUEST_SL' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'RUPD$' || '%'

UNION ALL

SELECT
	A.TABLE_NAME
	,C.COMMENTS AS TAB_COMMENTS
	,A.COLUMN_ID
	,A.COLUMN_NAME
	,A.DATA_TYPE
	,A.NULLABLE
	,B.COMMNETS
FROM	ALL_TAB_COMMENTS A
	,ALL_COL_COMMENTS B
	,ALL_TAB_COMMENTS C
WHERE	A.OWNER = B.OWNER
   AND	A.OWNER = C.OWNER
   AND	A.TABLE_NAME = B.TABLE_NAME (+)
   AND	A.TABLE_NAME = C.TABLE_NAME (+)
   AND	A.COLUMN_NAME = B.COLUMN_NAME (+)
   AND	A.OWNER = 'USER_PM'
   AND	A.TABLE_NAME NOT LIKE 'TAB' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'PLAN_TABLE' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'MLOG$' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'QUEST_SL' || '%'	
   AND	A.TABLE_NAME NOT LIKE 'RUPD$' || '%'
) A
ORDER BY A.TABLE_NAME, A.COLUMN_ID
